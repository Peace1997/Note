计算 URDF 整机质量：
```python
import xml.etree.ElementTree as ET 
def compute_urdf_mass(file_path): 
	tree = ET.parse(file_path) 
	root = tree.getroot() 
	total_mass = 0.0 
	for link in root.findall('link'): 
		inertial = link.find('inertial') 
		if inertial is not None: 
			mass = inertial.find('mass') 
			if mass is not None: 
				total_mass += float(mass.get('value')) 
	return total_mass 
if __name__ == "__main__": 
	urdf_file_path = 'robot.urdf' 
	total_mass = compute_urdf_mass(urdf_file_path) 
	print(f'Total mass of the robot is {total_mass} kg')
```
